1.1 函数

  • [ ] overloading
  • [X] 支持默认参数
  • [X] 多个函数值
  • [X] 变长参数
def echo(i1, i2, i3):
    return (i1, i2, i3)


def echo2(*args):
    return args


if __name__ == "__main__":
    print(echo(1, 2, 3))
    print(echo2(1, 2, 3, 4))

(1, 2, 3) (1, 2, 3, 4)
