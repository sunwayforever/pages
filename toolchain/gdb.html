1.3.4 call

# 使用 call 来调用函数
(gdb) call print_rtl_single(stdout, insn)

# call 的返回值
(gdb) call PATTERN(insn)
$10 = (rtx &) @0x7ffff7676920: 0x7ffff778a288
(gdb) p *$10
...

# 可以使用 $ 引用最近的返回值, 即 $10
# 使用 set $xxx 可以保存这个返回值到另一个名字
(gdb) set $pat=$
(gdb) p *$pat
